            function favoriteCheck() {
                $.ajax({
                    url: "https://fujoho.jp/index.php?p=fav_check_json&check=shop",
                    dataType: "json",
                    xhrFields: {
                        withCredentials: true
                    },
                    success: function(fav) {
                        $('[id^="fav_icon_off_"]').show();
                        $('[id^="fav_icon_on_"]').hide();
                        if (fav.msg !== 'error') {
                            for (var i = 0; i < fav.shop.length; i++) {
                                changeFavStarState(fav.shop[i], $.inArray(fav.shop[i], fav.shop));
                            }
                        }
                    },
                    error: function(req, sts, err) {}
                });
            }
